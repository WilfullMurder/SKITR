143 static json valueToJson(const core::Value& v) {
144 using core::Type;
145 switch (v.type()) {
146 case Type::Null:
147 return nullptr;
148 case Type::Bool:
149 return v.asBool();
150 case Type::Int:
151 return v.asInt();
152 case Type::Double:
153 return v.asDouble();
154 case Type::String:
155 return v.asString();
156 case Type::Array: {
157 json arr = json::array();
158 for (const auto &elem: v.asArray()) arr.push_back(valueToJson(elem));
159 return arr;
160 }
161 case Type::Object: {
162 json obj = json::object();
163 for (const auto &[key, val]: v.asObject()) obj[key] = valueToJson(val);
164 return obj;
165 }
166 default:
167 return nullptr; // should never happen
168 }

171 static core::Value jsonToValue(const json& j) {
172 using core::Value;
173 if (j.is_null()) return Value();
174 if (j.is_boolean()) return Value(j.get<bool>());
175 if (j.is_number_integer()) return Value(static_cast<int64_t>(j.get<int64_t>()));
176 if (j.is_number_unsigned()) return Value(static_cast<int64_t>(j.get<uint64_t>()));
177 if (j.is_number_float()) return Value(j.get<double>());
178 if (j.is_string()) return Value(j.get<std::string>());
179 if (j.is_array()) {
180 Value::array_t arr;
181 arr.reserve(j.size());
182 for (const auto &el: j) arr.push_back(jsonToValue(el));
183 return Value(std::move(arr));
184 }
185 if (j.is_object()) {
186 Value::object_t obj;
187 for (auto it = j.begin(); it != j.end(); ++it) {
188 obj.emplace(it.key(), jsonToValue(it.value()));
189 }
190 return Value(std::move(obj));
191 }
192 // fallback to null
193 return Value();
194 }
