193 void Value::writeToStream(std::ostringstream &os) const {
194 switch (type_) {
195 case Type::Null:
196 os << "null";
197 break;
198 case Type::Bool:
199 os << (std::get<bool>(storage_) ? "true" : "false");
200 break;
201 case Type::Int:
202 os << std::get<int64_t>(storage_);
203 break;
204 case Type::Double:
205 os << std::get<double>(storage_);
206 break;
207 case Type::String:
208 os << '"' << escapeString(std::get<std::string>(storage_)) << '"';
209 break;
210 case Type::Array: {
211 os << '[';
212 const auto &ptr = *std::get<array_ptr>(storage_);
213 for (size_t i = 0; i < ptr.size(); ++i) {
214 if (i > 0) os << ',';
215 os << ptr[i].toString();
216 }
217 os << ']';
218 break;
219 }
220 case Type::Object: {
221 os << '{';
222 const auto &ptr = *std::get<object_ptr>(storage_);
223 bool first = true;
224 for (const auto &kv: ptr) {
225 if (!first) os << ',';
226 first = false;
227 os << '"' << escapeString(kv.first) << "\":" << kv.second.toString();
228 }
229 os << '}';
230 break;
231 }
232 case Type::Binary: {
233 os << "\"<binary:" << std::get<binary_t>(storage_).size() << " bytes>\"";
234 break;
235 }
236 case Type::DateTime: {
237 auto tp = std::get<datetime_t>(storage_);
238 std::time_t t = std::chrono::system_clock::to_time_t(tp);
239 os << '"' << std::asctime(std::localtime(&t)) << "";
240 break;
241 }
242 }

245 std::string Value::escapeString(const std::string &s) {
246 std::string out;
247 out.reserve(s.size() + 4);
248 for (char c: s) {
249 switch (c) {
250 case '"': out += "\\\""; break;
251 case '\\': out += "\\\\"; break;
252 case '\n': out += "\\n"; break;
253 case '\r': out += "\\r"; break;
254 case '\t': out += "\\t"; break;
255 default: out += c; break;
256 }
257 }
258 return out;
259 }
